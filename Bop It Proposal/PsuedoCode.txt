# Constants & Globals
SET MAX_TIME         TO 5
SET NUM_ACTIONS      TO 3
SET LED_DURATION     
SET MOVEMENT_THRESHOLD 

setup()
{
    SET score    TO 0
    SET time     TO MAX_TIME
    SET gameOver TO true
    initializePins()
    initializeDisplay()
    initializeSound()
    SET greenLED TO LOW
    SET redLED   TO LOW
}

loop()
{
    WHILE true
        waitForNewGame()
        SET score    TO 0
        SET time     TO MAX_TIME
        SET greenLED TO LOW
        SET redLED   TO LOW
        update_display_Score(score)

        SET gameOver TO false
        WHILE gameOver is false
            SET currentAction TO generateRandom(1, NUM_ACTIONS)

            IF currentAction == 1 THEN
                playSound(ACCELEROMETER_PROMPT)
                SET result TO checkAccelerometer()
            ELSE IF currentAction == 2 THEN
                playSound(BUTTON_PROMPT)
                SET result TO checkPushButton()
            ELSE IF currentAction == 3 THEN
                playSound(JOYSTICK_PROMPT)
                SET result TO checkJoystick()
            END IF

            IF result == CORRECT THEN
                INCREMENT score BY 1
                lightLED(GREEN)
                IF score == 99 THEN
                    playSound(WIN)
                    update_display_Score(score)
                    SET gameOver TO true
                ELSE
                    playSound(SUCCESS)
                    update_display_Score(score)
                END IF
            ELSE IF result == TIMEOUT OR result == INCORRECT THEN
                lightLED(RED)
                playSound(LOSE)
                update_display_Score(score)
                SET gameOver TO true
            END IF
        END WHILE
    END WHILE
}

waitForNewGame()
{
    SET greenLED TO LOW
    SET redLED   TO LOW
    WHILE true
        READ newGameButton state
        IF newGameButton is pressed THEN
            RETURN
        END IF
    END WHILE
}

checkAccelerometer()
{
    SET time TO MAX_TIME
    WHILE time > 0
        decreaseTime()
        READ accelerometer magnitude
        READ button state
        READ joystick direction

        IF accelerometer magnitude > MOVEMENT_THRESHOLD THEN
            RETURN CORRECT
        ELSE IF button is pressed OR joystick moved THEN
            RETURN INCORRECT
        END IF
    END WHILE
    RETURN INCORRECT
}

checkPushButton()
{
    SET time TO MAX_TIME
    WHILE time > 0
        decreaseTime()
        READ button state
        READ accelerometer magnitude
        READ joystick direction

        IF button is pressed THEN
            RETURN CORRECT
        ELSE IF accelerometer magnitude > MOVEMENT_THRESHOLD OR joystick moved THEN
            RETURN INCORRECT
        END IF
    END WHILE
    RETURN INCORRECT
}

checkJoystick()
{
    SET time TO MAX_TIME
    WHILE time > 0
        decreaseTime()
        READ joystick direction
        READ button state
        READ accelerometer magnitude

        IF joystick moved in ANY direction THEN
            RETURN CORRECT
        ELSE IF button is pressed OR accelerometer magnitude > MOVEMENT_THRESHOLD THEN
            RETURN INCORRECT
        END IF
    END WHILE
    RETURN INCORRECT
}

decreaseTime()
{
    SET time TO time - time_decrement #haven't decided timing decrement yet           
}

lightLED(color)
{
    IF color == GREEN THEN
        SET greenLED TO HIGH
        WAIT LED_DURATION
        SET greenLED TO LOW
    ELSE IF color == RED THEN
        SET redLED TO HIGH
    END IF
}

update_display_Score(score)
{
    DISPLAY score on screen
}


playSound(type)
{
    IF type == SUCCESS              THEN play success audio
    IF type == WIN                  THEN play win audio
    IF type == LOSE                 THEN play lose audio
    IF type == ACCELEROMETER_PROMPT THEN play accelerometer prompt audio
    IF type == BUTTON_PROMPT        THEN play button prompt audio
    IF type == JOYSTICK_PROMPT      THEN play joystick prompt audio
}

initializePins()
{
    SET pins to i/o and initial states
}

initializeDisplay()
{
    clear screen and set it up
}

initializeSound()
{
    initialize speaker and load or map sound types
}
